<Tabs>
    <Tab value="sandbox" title="cURL" default>

```jsx
curl -X GET \
  'https://api-dev-sandbox.kidapay.com/v1/orders?take=10&page=1%20' \
  -H 'token: TOKEN_FROM_ADMIN_PORTAL'

```

  </Tab>

  <Tab value="node" title="Node">

```js
const https = require("https");

const token = "TOKEN_FROM_ADMIN_PORTAL"; // Replace with your actual token

const options = {
  hostname: "api-dev-sandbox.kidapay.com",
  path: "/v1/orders?take=10&page=1%20",
  method: "GET",
  headers: {
    token: token,
  },
};

const req = https.request(options, (res) => {
  let data = "";

  console.log(`Status Code: ${res.statusCode}`);

  res.on("data", (chunk) => {
    data += chunk;
  });

  res.on("end", () => {
    try {
      const json = JSON.parse(data);
      console.log("Response:", json);
    } catch (err) {
      console.error("Failed to parse JSON:", data);
    }
  });
});

req.on("error", (err) => {
  console.error("Request Error:", err.message);
});

req.end();
```

  </Tab>  
  <Tab value="php" title="PHP">

```php
<?php

$token = 'TOKEN_FROM_ADMIN_PORTAL'; // Replace with your actual token
$url = 'https://api-dev-sandbox.kidapay.com/v1/orders?take=10&page=1%20';

$ch = curl_init();

curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    "token: $token"
]);

$response = curl_exec($ch);

if (curl_errno($ch)) {
    echo 'Request Error: ' . curl_error($ch);
} else {
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    echo "HTTP Status Code: $httpCode\n";

    $data = json_decode($response, true);
    echo "Response:\n";
    print_r($data);
}

curl_close($ch);

```
